## Goal

A single `@clickhouse/client` instance shared across the collector app. Imported by ingest and consumer code paths; not used by the migration runner (which creates its own client).

## Acceptance criteria

- `import { ch } from '../db/clickhouse.ts'` returns a connected, ready client.
- Reads URL/user/password/database from `src/config.ts`.
- Exports a typed `insertEvents(rows)` helper used by the consumer.
- Exports a typed `query` helper for read paths (returns parsed JSON arrays).
- Lazy-initialized: client is constructed on first import, not at module load (so tests can mock).
- Has a `close()` for graceful shutdown.

## Implementation notes

```ts
// apps/collector/src/db/clickhouse.ts (sketch)
import { createClient, type ClickHouseClient } from '@clickhouse/client';
import { config } from '../config.ts';

let _client: ClickHouseClient | null = null;

function client(): ClickHouseClient {
  if (!_client) {
    _client = createClient({
      url: config.clickhouse.url,
      username: config.clickhouse.user,
      password: config.clickhouse.password,
      database: config.clickhouse.database,
      compression: { request: true, response: true },
      keep_alive: { enabled: true },
      request_timeout: 30_000,
    });
  }
  return _client;
}

/** Insert rows into the events_buffer table (JSONEachRow). */
export async function insertEvents(rows: object[]): Promise<void> {
  await client().insert({
    table: 'events_buffer',
    values: rows,
    format: 'JSONEachRow',
  });
}

/** Run a read query and return rows. */
export async function query<T = unknown>(
  q: string,
  params?: Record<string, unknown>,
): Promise<T[]> {
  const result = await client().query({
    query: q,
    query_params: params,
    format: 'JSONEachRow',
  });
  return (await result.json()) as T[];
}

export async function close(): Promise<void> {
  if (_client) {
    await _client.close();
    _client = null;
  }
}
```

## Tests

Smoke test under `bun test`:

```ts
// apps/collector/src/db/__tests__/clickhouse.test.ts
import { describe, expect, it } from 'bun:test';
import { query } from '../clickhouse.ts';

describe('clickhouse client', () => {
  // Skip if no CH locally; CI provides a service container.
  it.skipIf(!process.env.CLICKHOUSE_URL)('runs SELECT 1', async () => {
    const rows = await query<{ one: number }>('SELECT 1 AS one');
    expect(rows[0]?.one).toBe(1);
  });
});
```

## Out of scope

- Per-request retry logic — that lives in the consumer (task 1.5).
- Connection pooling — `@clickhouse/client` over HTTP doesn't pool but reuses keep-alive; sufficient for v1.
- Read replicas / sharding.
